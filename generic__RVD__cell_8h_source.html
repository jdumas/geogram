288  template <index_t DIM>
289  signed_index_t clip_by_plane(
290  const Mesh* mesh, const Delaunay* delaunay,
291  index_t i, index_t j,
292  bool exact, bool symbolic
293  ) {
294  index_t new_v = create_vertex();
295  set_vertex_id(index_t(new_v), signed_index_t(j)+1);
296  index_t conflict_begin, conflict_end;
297 
298  // Phase I: Determine the conflict zone and chain the triangles
299  // Note: they are not immediately deleted, since we need the
300  // geometric information in the triangles to compute the new
301  // intersections.
302  get_conflict_list<DIM>(
303  mesh, delaunay, i, j, exact, conflict_begin, conflict_end
304  );
305 
306  // Special case: the clipping plane did not clip anything.
307  if(conflict_begin == END_OF_LIST) {
308  return signed_index_t(new_v);
309  }
310 
311  // Phase II: Find a triangle on the border of the conflict zone
312  // (by traversing the conflict list).
313  index_t first_conflict_t;
314  index_t first_conflict_e;
315  bool found_h = find_triangle_on_border(
316  conflict_begin, conflict_end,
317  first_conflict_t, first_conflict_e
318  );
319 
320  // The clipping plane removed everything !
321  // (note: cannot be empty conflict list, since this
322  // case was detected by previous test at the end of
323  // phase I)
324  if(!found_h) {
325  clear();
326  return -1;
327  }
328 
329  // Phase III: Triangulate hole.
330  triangulate_hole<DIM>(
331  delaunay, i, j, symbolic,
332  first_conflict_t, first_conflict_e,
333  new_v
334  );
335 
336  // Phase IV: Merge the conflict zone into the free list.
337  merge_into_free_list(conflict_begin, conflict_end);
338  return signed_index_t(new_v);
339  }

893  template <index_t DIM>
894  index_t triangulate_hole(
895  const Delaunay* delaunay,
896  index_t i, index_t j, bool symbolic,
897  index_t t1, index_t t1ebord,
898  index_t v_in
899  ) {
900  index_t t = t1;
901  index_t e = t1ebord;
902  index_t t_adj = triangle_adjacent(t,e);
903  geo_debug_assert(t_adj != index_t(-1));
904 
905  geo_debug_assert(triangle_is_conflict(t));
906  geo_debug_assert(!triangle_is_conflict(t_adj));
907 
908  index_t new_t_first = index_t(-1);
909  index_t new_t_prev = index_t(-1);
910 
911  do {
912 
913  index_t v1 = triangle_vertex(t, plus1mod3(e));
914  index_t v2 = triangle_vertex(t, minus1mod3(e));
915 
916  // Create new triangle
917  index_t new_t = create_triangle(v_in, v1, v2);
918 
919  triangle_dual(new_t).intersect_geom<DIM>(
920  intersections_,
921  triangle_dual(t),
922  triangle_dual(triangle_adjacent(t, e)),
923  delaunay->vertex_ptr(i), delaunay->vertex_ptr(j)
924  );
925 
926  if(symbolic) {
927  triangle_dual(new_t).sym().intersect_symbolic(
928  triangle_dual(t).sym(),
929  triangle_dual(triangle_adjacent(t, e)).sym(),
930  j
931  );
932  }
933 
934  // Connect new triangle to triangle on the other
935  // side of the conflict zone.
936  set_triangle_adjacent(new_t, 0, t_adj);
937  index_t adj_e = triangle_adjacent_index(t_adj, t);
938  set_triangle_adjacent(t_adj, adj_e, new_t);
939 
940 
941  // Move to next triangle
942  e = plus1mod3(e);
943  t_adj = index_t(triangle_adjacent(t,e));
944  while(triangle_is_conflict(t_adj)) {
945  t = t_adj;
946  e = minus1mod3(find_triangle_vertex(t,v2));
947  t_adj = index_t(triangle_adjacent(t,e));
948  geo_debug_assert(t_adj != index_t(-1));
949  }
950 
951  if(new_t_prev == index_t(-1)) {
952  new_t_first = new_t;
953  } else {
954  set_triangle_adjacent(new_t_prev, 1, new_t);
955  set_triangle_adjacent(new_t, 2, new_t_prev);
956  }
957 
958  new_t_prev = new_t;
959 
960  } while((t != t1) || (e != t1ebord));
961 
962  // Connect last triangle to first triangle
963  set_triangle_adjacent(new_t_prev, 1, new_t_first);
964  set_triangle_adjacent(new_t_first, 2, new_t_prev);
965 
966  return new_t_prev;
967  }

986  template <index_t DIM>
987  void get_conflict_list(
988  const Mesh* mesh, const Delaunay* delaunay,
989  index_t i, index_t j, bool exact,
990  index_t& conflict_begin, index_t& conflict_end
991  ) {
992  conflict_begin = END_OF_LIST;
993  conflict_end = END_OF_LIST;
994  if(exact) {
995  // In exact mode, we classify each vertex
996  // using the exact predicate. Note that
997  // "climbing/walking" from a random vertex
998  // would be more efficient, but it would require a
999  // "comparison" exact predicate (with 8 different
1000  // versions according to the configuration of the
1001  // two vertices to be compared)
1002  for(index_t t = 0; t < max_t(); t++) {
1003  if(triangle_is_used(t)) {
1004  Sign s = side<DIM>(
1005  mesh, delaunay,
1006  triangle_dual(t),
1007  i, j,
1008  exact
1009  );
1010  if(s == GEO::NEGATIVE) {
1011  append_triangle_to_conflict_list(
1012  t, conflict_begin, conflict_end
1013  );
1014  }
1015  }
1016  }
1017  } else {
1018  // In non-exact mode, we first detect the
1019  // vertex that is furthest away along the
1020  // normal vector of the clipping bisector,
1021  // and then we propagate using a flood-fill
1022  // algorithm. This strategy ensures that
1023  // the conflict zone remains connected, even
1024  // in the presence of numerical errors. Clearly
1025  // it does not ensure validity, but in practice
1026  // it improves resistance to degeneracies.
1027  index_t furthest_t =
1028  find_furthest_point_linear_scan<DIM>(
1029  delaunay, i, j
1030  );
1031  propagate_conflict_list<DIM>(
1032  mesh, delaunay, furthest_t,
1033  i, j, exact,
1034  conflict_begin, conflict_end
1035  );
1036  }
1037  }

1050  template <index_t DIM>
1051  index_t find_furthest_point_linear_scan(
1052  const Delaunay* delaunay, index_t i, index_t j
1053  ) const {
1054  index_t result = NO_TRIANGLE;
1055  double furthest_dist = 0.0;
1056  for(index_t t = 0; t < max_t(); ++t) {
1057  if(triangle_is_used(t)) {
1058  double d = signed_bisector_distance<DIM>(
1059  delaunay, i, j, triangle_dual(t).point()
1060  );
1061  if(d < furthest_dist) {
1062  result = t;
1063  furthest_dist = d;
1064  }
1065  }
1066  }
1067  return (furthest_dist < 0) ? result : NO_TRIANGLE;
1068  }

1081  template <index_t DIM>
1082  static double signed_bisector_distance(
1083  const Delaunay* delaunay, index_t i, index_t j, const double* q
1084  ) {
1085  const double* pi = delaunay->vertex_ptr(i);
1086  const double* pj = delaunay->vertex_ptr(j);
1087  double result = 0;
1088  for(coord_index_t c = 0; c < DIM; ++c) {
1089  result += GEO::geo_sqr(q[c] - pj[c]);
1090  result -= GEO::geo_sqr(q[c] - pi[c]);
1091  }
1092  return result;
1093  }
