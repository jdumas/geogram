668  inline Sign orient_3d_inexact(
669  const double* p0, const double* p1,
670  const double* p2, const double* p3
671  ) {
672  double a11 = p1[0] - p0[0] ;
673  double a12 = p1[1] - p0[1] ;
674  double a13 = p1[2] - p0[2] ;
675 
676  double a21 = p2[0] - p0[0] ;
677  double a22 = p2[1] - p0[1] ;
678  double a23 = p2[2] - p0[2] ;
679 
680  double a31 = p3[0] - p0[0] ;
681  double a32 = p3[1] - p0[1] ;
682  double a33 = p3[2] - p0[2] ;
683 
684  double Delta = det3x3(
685  a11,a12,a13,
686  a21,a22,a23,
687  a31,a32,a33
688  );
689 
690  return geo_sgn(Delta);
691  }
