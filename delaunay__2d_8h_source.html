797  bool triangle_is_conflict(index_t t, const double* p) const {
798 
799  // Lookup triangle vertices
800  const double* pv[3];
801  for(index_t i=0; i<3; ++i) {
802  signed_index_t v = triangle_vertex(t,i);
803  pv[i] = (v == -1) ? nullptr : vertex_ptr(index_t(v));
804  }
805 
806  // Check for virtual triangles (then in_circle()
807  // is replaced with orient2d())
808  for(index_t le = 0; le < 3; ++le) {
809 
810  if(pv[le] == nullptr) {
811 
812  // Facet of a virtual triangle opposite to
813  // infinite vertex corresponds to
814  // the triangle on the convex hull of the points.
815  // Orientation is obtained by replacing vertex lf
816  // with p.
817  pv[le] = p;
818  Sign sign = PCK::orient_2d(pv[0],pv[1],pv[2]);
819 
820  if(sign > 0) {
821  return true;
822  }
823 
824  if(sign < 0) {
825  return false;
826  }
827 
828  // If sign is zero, we check the real triangle
829  // adjacent to the facet on the convex hull.
830  geo_debug_assert(triangle_adjacent(t, le) >= 0);
831  index_t t2 = index_t(triangle_adjacent(t, le));
832  geo_debug_assert(!triangle_is_virtual(t2));
833 
834  // If t2 is already chained in the conflict list,
835  // then it is conflict
836  if(triangle_is_in_list(t2)) {
837  return true;
838  }
839 
840  // If t2 is marked, then it is not in conflict.
841  if(triangle_is_marked(t2)) {
842  return false;
843  }
844 
845  return triangle_is_conflict(t2, p);
846  }
847  }
848 
849  // If the triangle is a finite one, it is in conflict
850  // if its circumscribed sphere contains the point (this is
851  // the standard case).
852 
853  if(weighted_) {
854  double h0 = heights_[finite_triangle_vertex(t, 0)];
855  double h1 = heights_[finite_triangle_vertex(t, 1)];
856  double h2 = heights_[finite_triangle_vertex(t, 2)];
857  index_t pindex = index_t(
858  (p - vertex_ptr(0)) / int(vertex_stride_)
859  );
860  double h = heights_[pindex];
861  return (PCK::orient_2dlifted_SOS(
862  pv[0],pv[1],pv[2],p,h0,h1,h2,h
863  ) > 0) ;
864  }
865 
866  return (PCK::in_circle_2d_SOS(pv[0], pv[1], pv[2], p) > 0);
867  }

879  static index_t find_3(const signed_index_t* T, signed_index_t v) {
880  // The following expression is 10% faster than using
881  // if() statements. This uses the C++ norm, that
882  // ensures that the 'true' boolean value converted to
883  // an int is always 1. With most compilers, this avoids
884  // generating branching instructions.
885  // Thank to Laurent Alonso for this idea.
886  index_t result = index_t( (T[1] == v) | ((T[2] == v) * 2) );
887  // Sanity check, important if it was T[0], not explicitly
888  // tested (detects input that does not meet the precondition).
889  geo_debug_assert(T[result] == v);
890  return result;
891  }
