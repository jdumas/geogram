298  index_t array_size(index_t array) const {
299  return size_[array];
300  }
301 
311  index_t array_capacity(index_t array) const {
312  index_t size = array_size(array);
313  if(size == 0) {
314  return 0;
315  }
316  index_t result = 1;
317  index_t mask = 1;
318  for(index_t i = 0; i < 32; i++) {
319  mask = mask << 1;
320  if((size & mask) != 0) {
321  result = mask;
322  }
323  }
324  // If size is not already a power of two,
325  if(result != size) {
326  result = result << 1;
327  }
328  return result;
329  }
330 
337  signed_index_t find_index(index_t array, index_t key) const {
338  index_t* K = keys_[array];
339  for(index_t i = 0; i < array_size(array); ++i) {
340  if(K[i] == key) {
341  return signed_index_t(i);
342  }
343  }
344  return -1;
345  }
346 
355  signed_index_t find_value(index_t array, index_t key) const {
356  signed_index_t i = find_index(array, key);
357  if(i == -1) {
358  return -1;
359  }
360  return signed_index_t(values_[array][i]);
361  }
362 
369  void insert(index_t array, index_t key, index_t value) {
370  signed_index_t si = find_index(array, key);
371  if(si == -1) {
372  // If not found, append at the end of array
373  index_t i = size_[array];
374  if(i == array_capacity(array)) {
375  // If capacity is reached, grow storage
376  index_t new_nb = index_t(2*i);
377  if(new_nb == 0) {
378  new_nb = 1;
379  }
380  keys_[array] = reinterpret_cast<index_t*>(
381  realloc(keys_[array], sizeof(index_t) * new_nb)
382  );
383  values_[array] = reinterpret_cast<index_t*>(
384  realloc(values_[array], sizeof(index_t) * new_nb)
385  );
386  }
387  size_[array] = i + 1;
388  si = signed_index_t(i);
389  }
390  keys_[array][si] = key;
391  values_[array][si] = value;
392  }
393 
394  private:
395  std::vector<index_t*> keys_;
396  std::vector<index_t*> values_;
397  std::vector<index_t> size_;
398  };
