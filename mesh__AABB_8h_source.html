101  void bbox_intersect_recursive(
102  std::function<void(index_t)> action, const Box& box,
103  index_t node, index_t b, index_t e
104  ) const {
105  geo_debug_assert(e != b);
106 
107  // Prune sub-tree that does not have intersection
108  if(!bboxes_overlap(box, bboxes_[node])) {
109  return;
110  }
111 
112  // Leaf case
113  if(e == b+1) {
114  action(b);
115  return;
116  }
117 
118  // Recursion
119  index_t m = b + (e - b) / 2;
120  index_t node_l = 2 * node;
121  index_t node_r = 2 * node + 1;
122 
123  bbox_intersect_recursive(action, box, node_l, b, m);
124  bbox_intersect_recursive(action, box, node_r, m, e);
125  }

148  void self_intersect_recursive(
149  std::function<void(index_t,index_t)> action,
150  index_t node1, index_t b1, index_t e1,
151  index_t node2, index_t b2, index_t e2
152  ) const {
153  geo_debug_assert(e1 != b1);
154  geo_debug_assert(e2 != b2);
155 
156  // Since we are intersecting the AABBTree with *itself*,
157  // we can prune half of the cases by skipping the test
158  // whenever node2's facet index interval is greated than
159  // node1's facet index interval.
160  if(e2 <= b1) {
161  return;
162  }
163 
164  // The acceleration is here:
165  if(!bboxes_overlap(bboxes_[node1], bboxes_[node2])) {
166  return;
167  }
168 
169  // Simple case: leaf - leaf intersection.
170  if(b1 + 1 == e1 && b2 + 1 == e2) {
171  action(b1, b2);
172  return;
173  }
174 
175  // If node2 has more elements than node1, then
176  // intersect node2's two children with node1
177  // else
178  // intersect node1's two children with node2
179  if(e2 - b2 > e1 - b1) {
180  index_t m2 = b2 + (e2 - b2) / 2;
181  index_t node2_l = 2 * node2;
182  index_t node2_r = 2 * node2 + 1;
183  self_intersect_recursive(action, node1, b1, e1, node2_l, b2, m2);
184  self_intersect_recursive(action, node1, b1, e1, node2_r, m2, e2);
185  } else {
186  index_t m1 = b1 + (e1 - b1) / 2;
187  index_t node1_l = 2 * node1;
188  index_t node1_r = 2 * node1 + 1;
189  self_intersect_recursive(action, node1_l, b1, m1, node2, b2, e2);
190  self_intersect_recursive(action, node1_r, m1, e1, node2, b2, e2);
191  }
192  }

272  static index_t max_node_index(index_t node_index, index_t b, index_t e) {
273  geo_debug_assert(e > b);
274  if(b + 1 == e) {
275  return node_index;
276  }
277  index_t m = b + (e - b) / 2;
278  index_t childl = 2 * node_index;
279  index_t childr = 2 * node_index + 1;
280  return std::max(
281  max_node_index(childl, b, m),
282  max_node_index(childr, m, e)
283  );
284  }

296  void init_bboxes_recursive(
297  index_t node_index,
298  index_t b, index_t e,
299  std::function<void(Box&, index_t)> get_bbox
300  ) {
301  geo_debug_assert(node_index < bboxes_.size());
302  geo_debug_assert(b != e);
303  if(b + 1 == e) {
304  get_bbox(bboxes_[node_index], b);
305  return;
306  }
307  index_t m = b + (e - b) / 2;
308  index_t childl = 2 * node_index;
309  index_t childr = 2 * node_index + 1;
310  geo_debug_assert(childl < bboxes_.size());
311  geo_debug_assert(childr < bboxes_.size());
312  init_bboxes_recursive(childl, b, m, get_bbox);
313  init_bboxes_recursive(childr, m, e, get_bbox);
314  geo_debug_assert(childl < bboxes_.size());
315  geo_debug_assert(childr < bboxes_.size());
316  bbox_union(bboxes_[node_index], bboxes_[childl], bboxes_[childr]);
317  }

320  index_t nb_;
321  vector<Box> bboxes_;
322  };
