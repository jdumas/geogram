385  bool compute_inverse(matrix_type& result) const {
386  FT val=FT(0.0), val2=FT(0.0);
387  matrix_type tmp = (*this);
388 
389  result.load_identity();
390 
391  for(index_t i = 0; i != DIM; i++) {
392  val = tmp(i, i); /* find pivot */
393  index_t ind = i;
394  for(index_t j = i + 1; j != DIM; j++) {
395  if(fabs(tmp(j, i)) > fabs(val)) {
396  ind = j;
397  val = tmp(j, i);
398  }
399  }
400 
401  if(ind != i) {
402  for(index_t j = 0; j != DIM; j++) {
403  val2 = result(i, j);
404  result(i, j) = result(ind, j);
405  result(ind, j) = val2; /* swap columns */
406  val2 = tmp(i, j);
407  tmp(i, j) = tmp(ind, j);
408  tmp(ind, j) = val2;
409  }
410  }
411 
412  if(val == 0.0) {
413  return false;
414  }
415 
416  for(index_t j = 0; j != DIM; j++) {
417  tmp(i, j) /= val;
418  result(i, j) /= val;
419  }
420 
421  for(index_t j = 0; j != DIM; j++) {
422  if(j == i) {
423  continue; /* eliminate column */
424  }
425  val = tmp(j, i);
426  for(index_t k = 0; k != DIM; k++) {
427  tmp(j, k) -= tmp(i, k) * val;
428  result(j, k) -= result(i, k) * val;
429  }
430  }
431  }
432 
433  return true;
434  }
