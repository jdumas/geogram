76  inline bool is_valid(const vector<index_t>& permutation) {
77  std::vector<bool> visited(permutation.size(), false);
78  for(index_t i = 0; i < permutation.size(); i++) {
79  if(permutation[i] >= permutation.size()) {
80  return false;

82  if(visited[permutation[i]]) {

84  }
85  visited[permutation[i]] = true;

163  inline void apply(
164  void* data, const vector<index_t>& permutation_in,
165  index_t elemsize
166  ) {
167  Memory::pointer pdata = (Memory::pointer) (data);
168  vector<index_t>& permutation =
169  const_cast<vector<index_t>&>(permutation_in);
170  geo_debug_assert(is_valid(permutation));
171  Memory::byte* temp = static_cast<Memory::byte*>(alloca(elemsize));
172  for(index_t k = 0; k < permutation.size(); k++) {
173  if(is_marked(permutation, k)) {
174  continue;
175  }
176  index_t i = k;
177  index_t j = permutation[k];
178  Memory::copy(temp, pdata + i * elemsize, elemsize);
179  mark(permutation, k);
180  while(j != k) {
181  Memory::copy(
182  pdata + i * elemsize, pdata + j * elemsize, elemsize
183  );
184  index_t nj = permutation[j];
185  mark(permutation, j);
186  i = j;
187  j = nj;
188  }
189  Memory::copy(pdata + i * elemsize, temp, elemsize);
190  }
191  for(index_t k = 0; k < permutation.size(); k++) {
192  unmark(permutation, k);
193  }
194  }

259  inline void invert(vector<index_t>& permutation) {
260  geo_debug_assert(is_valid(permutation));
261  for(index_t k = 0; k < permutation.size(); k++) {
262  if(is_marked(permutation, k)) {
263  continue;
264  }
265  index_t i = k;
266  index_t j = permutation[i];
267  while(j != k) {
268  index_t temp = permutation[j];
269  permutation[j] = i;
270  mark(permutation, j);
271  i = j;
272  j = temp;
273  }
274  permutation[j] = i;
275  mark(permutation, j);
276  }
277  for(index_t k = 0; k < permutation.size(); k++) {
278  unmark(permutation, k);
279  }
