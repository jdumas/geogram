271  inline void* aligned_malloc(
272  size_t size, size_t alignment = GEO_MEMORY_ALIGNMENT
273  ) {
274 #if defined(GEO_OS_ANDROID)
275  // Alignment not supported under Android.
276  geo_argused(alignment);
277  return malloc(size);
278 #elif defined(GEO_COMPILER_INTEL)
279  return _mm_malloc(size, alignment);
280 #elif defined(GEO_COMPILER_GCC) || defined(GEO_COMPILER_CLANG)
281  void* result;
282  return posix_memalign(&result, alignment, size) == 0
283  ? result : nullptr;
284 #elif defined(GEO_COMPILER_MSVC)
285  return _aligned_malloc(size, alignment);
286 #else
287  geo_argused(alignment);
288  return malloc(size);
289 #endif

418  inline void* align(void* p) {
419  size_t offset = (
420  GEO_MEMORY_ALIGNMENT -
421  (reinterpret_cast<size_t>(p) & (GEO_MEMORY_ALIGNMENT - 1))
422  ) & (GEO_MEMORY_ALIGNMENT - 1);
423  return reinterpret_cast<char*>(p) + offset;
424  }
