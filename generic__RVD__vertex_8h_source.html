359  template <class T, index_t DIM1, index_t DIM2, index_t DIM3>
360  inline void sets_intersect(
361  const small_set<T, DIM1>& S1,
362  const small_set<T, DIM2>& S2,
363  small_set<T, DIM3>& I
364  ) {
365  I.clear();
366  auto i1 = S1.begin();
367  auto i2 = S2.begin();
368  while(i1 < S1.end() && i2 < S2.end()) {
369  if(*i1 < *i2) {
370  ++i1;
371  }
372  else if(*i2 < *i1) {
373  ++i2;
374  }
375  else {
376  I.push_back(*i1);
377  ++i1;
378  ++i2;
379  }
380  }

588  bool intersect_symbolic(
589  const thisclass& v1,
590  const thisclass& v2,
591  index_t E
592  ) {
593 
594  // Compute the symbolic representation as the intersection
595  // of three planes.
596  sets_intersect(v1, v2, *this);
597  // this computes the set of planes that contain
598  // the edge [v1,v2]
599 
600  add_bisector(E); // the intersection is on E.
601 
602  // Compute the symbolic representation as intersection between
603  // bisector and boundary edge
604  // (it's redundant and less elegant than the representation
605  // as planes interactions,
606  // but we need this to handle degenerate configurations properly,
607  // and to use exact predicates with original boundary vertices
608  // coordinates).
609 
610  if(nb_boundary_facets() == 2) {
611  // If *this is on the intersection of two boundary facets,
612  // then *this is on
613  // a boundary edge, and we need to retrieve the indices of the
614  // two extremities of this boundary edge.
615 
616  index_t nb1 = v1.nb_boundary_facets();
617  index_t nb2 = v2.nb_boundary_facets();
618  if(nb1 == 3 && nb2 == 3) {
619  // If v1 and v2 are boundary vertices,
620  // then I is on the boundary
621  // edge that connects v1 and v2
622  set_boundary_edge(
623  v1.get_boundary_vertex(),
624  v2.get_boundary_vertex()
625  );
626  } else if(nb1 == 2) {
627  geo_debug_assert(nb_boundary_facets() == 2);
628  // If v1 is on a boundary edge,
629  // then I is on the same boundary edge as v1
630  copy_boundary_edge_from(v1);
631  } else if(nb2 == 2) {
632  geo_debug_assert(nb_boundary_facets() == 2);
633  // If v2 is on a boundary edge,
634  // then I is on the same boundary edge as v2
635  copy_boundary_edge_from(v2);
636  }
637  }
638 
639  // Sanity check: problem detected here, we
640  // notify the caller that will use a workaround
641  // (see clip_by_plane())
642  if(baseclass::size() != 3) {
643  return false;
644  }
645  return true;
646  }
647 
648  private:
649  index_t v1_;
650  index_t v2_;
651  };

982  template <index_t DIM>
983  void intersect_geom(
984  PointAllocator& target_intersections,
985  const Vertex& vq1, const Vertex& vq2,
986  const double* p1, const double* p2
987  ) {
988  const double* q1 = vq1.point();
989  const double* q2 = vq2.point();
990  double* Ipoint = target_intersections.new_item();
991  set_point(Ipoint);
992  double d = 0.0, l1 = 0.0, l2 = 0.0;
993  for(coord_index_t c = 0; c < DIM; ++c) {
994  double n = p1[c] - p2[c];
995  d -= n * (p2[c] + p1[c]);
996  l1 += q2[c] * n;
997  l2 += q1[c] * n;
998  }
999  d = 0.5 * d;
1000  l1 = ::fabs(l1 + d);
1001  l2 = ::fabs(l2 + d);
1002  double l12 = l1 + l2;
1003  if(l12 > 1e-30) {
1004  l1 /= l12;
1005  l2 /= l12;
1006  } else {
1007  l1 = 0.5;
1008  l2 = 0.5;
1009  }
1010  for(coord_index_t c = 0; c < DIM; ++c) {
1011  Ipoint[c] = l1 * q1[c] + l2 * q2[c];
1012  }
1013  set_weight(l1 * vq1.weight() + l2 * vq2.weight());
1014  }

1026  template <index_t DIM>

1036  }
