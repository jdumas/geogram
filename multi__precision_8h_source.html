81  inline void two_sum(double a, double b, double& x, double& y) {
82  x = a + b;
83  double bvirt = x - a;
84  double avirt = x - bvirt;
85  double bround = b - bvirt;
86  double around = a - avirt;
87  y = around + bround;
88  }

100  x = a - b;

102  double avirt = x + bvirt;
103  double bround = bvirt - b;
104  double around = a - avirt;
105  y = around + bround;
106  }

117  inline void split(double a, double& ahi, double& alo) {
118  double c = expansion_splitter_ * a;
119  double abig = c - a;
120  ahi = c - abig;
121  alo = a - ahi;

133  inline void two_product(double a, double b, double& x, double& y) {
134 #ifdef FP_FAST_FMA
135  // If the target processor supports the FMA (Fused Multiply Add)
136  // instruction, then the product of two doubles into a length-2
137  // expansion can be implemented as follows. Thanks to Marc Glisse
138  // for the information.
139  // Note: under gcc, automatic generations of fma() for a*b+c needs
140  // to be deactivated, using -ffp-contract=off, else it may break
141  // other functions such as fast_expansion_sum_zeroelim().
142  x = a*b;
143  y = fma(a,b,-x);
144 #else
145  x = a * b;
146  double ahi, alo;
147  split(a, ahi, alo);
148  double bhi, blo;
149  split(b, bhi, blo);
150  double err1 = x - (ahi * bhi);
151  double err2 = err1 - (alo * bhi);
152  double err3 = err2 - (ahi * blo);
153  y = (alo * blo) - err3;
154 #endif
155  }

165  inline void square(double a, double& x, double& y) {
166 #ifdef FP_FAST_FMA
167  // If the target processor supports the FMA (Fused Multiply Add)
168  // instruction, then the product of two doubles into a length-2
169  // expansion can be implemented as follows. Thanks to Marc Glisse
170  // for the information.
171  // Note: under gcc, automatic generations of fma() for a*b+c needs
172  // to be deactivated, using -ffp-contract=off, else it may break
173  // other functions such as fast_expansion_sum_zeroelim().
174  x = a*a;
175  y = fma(a,a,-x);
176 #else
177  x = a * a;
178  double ahi, alo;
179  split(a, ahi, alo);
180  double err1 = x - (ahi * ahi);
181  double err3 = err1 - ((ahi + ahi) * alo);
182  y = (alo * alo) - err3;
183 #endif
184  }
