331  bool get_neighbor_along_conflict_zone_border(
332  index_t t1,
333  index_t t1fborder,
334  index_t t1ft2,
335  index_t& t2,
336  index_t& t2fborder,
337  index_t& t2ft1
338  ) const {
339 
340  // Note: this function is a bit long for an inline function,
341  // but I observed a (modest) performance gain doing so.
342 
343  // Find two vertices that are both on facets new_f and f1
344  // (the edge around which we are turning)
345  // This uses duality as follows:
346  // Primal form (not used here):
347  // halfedge_facet_[v1][v2] returns a facet that is incident
348  // to both v1 and v2.
349  // Dual form (used here):
350  // halfedge_facet_[f1][f2] returns a vertex that both
351  // f1 and f2 are incident to.
352  signed_index_t ev1 =
353  tet_vertex(t1, index_t(halfedge_facet_[t1ft2][t1fborder]));
354  signed_index_t ev2 =
355  tet_vertex(t1, index_t(halfedge_facet_[t1fborder][t1ft2]));
356 
357  // Turn around edge [ev1,ev2] inside the conflict zone
358  // until we reach again the boundary of the conflict zone.
359  // Traversing inside the conflict zone is faster (as compared
360  // to outside) since it traverses a smaller number of tets.
361  index_t cur_t = t1;
362  index_t cur_f = t1ft2;
363  index_t next_t = index_t(tet_adjacent(cur_t,cur_f));
364  while(tet_is_in_list(next_t)) {
365  geo_debug_assert(next_t != t1);
366  cur_t = next_t;
367  cur_f = get_facet_by_halfedge(cur_t,ev1,ev2);
368  next_t = index_t(tet_adjacent(cur_t, cur_f));
369  }
370 
371  // At this point, cur_t is in conflict zone and
372  // next_t is outside the conflict zone.
373  index_t f12,f21;
374  get_facets_by_halfedge(next_t, ev1, ev2, f12, f21);
375  t2 = index_t(tet_adjacent(next_t,f21));
376  signed_index_t v_neigh_opposite = tet_vertex(next_t,f12);
377  t2ft1 = find_tet_vertex(t2, v_neigh_opposite);
378  t2fborder = cur_f;
379 
380  // Test whether the found neighboring tet was created
381  // (then return true) or is an old tet in conflict
382  // (then return false).
383  return(t2 != cur_t);
384  }

598  index_t new_tetrahedron() {
599  index_t result;
600  if(first_free_ == END_OF_LIST) {
601  cell_to_v_store_.resize(cell_to_v_store_.size() + 4, -1);
602  cell_to_cell_store_.resize(cell_to_cell_store_.size() + 4, -1);
603  // index_t(NOT_IN_LIST) is necessary, else with
604  // NOT_IN_LIST alone the compiler tries to generate a
605  // reference to NOT_IN_LIST resulting in a link error.
606  cell_next_.push_back(index_t(NOT_IN_LIST));
607  result = max_t() - 1;
608  } else {
609  result = first_free_;
610  first_free_ = tet_next(first_free_);
611  remove_tet_from_list(result);
612  }
613 
614  cell_to_cell_store_[4 * result] = -1;
615  cell_to_cell_store_[4 * result + 1] = -1;
616  cell_to_cell_store_[4 * result + 2] = -1;
617  cell_to_cell_store_[4 * result + 3] = -1;
618 
619  return result;
620  }

896  void get_facets_by_halfedge(
897  index_t t, signed_index_t v1, signed_index_t v2,
898  index_t& f12, index_t& f21
899  ) const {
900  geo_debug_assert(t < max_t());
901  geo_debug_assert(v1 != v2);
902 
903  // Find local index of v1 and v2 in tetrahedron t
904  // The following expression is 10% faster than using
905  // if() statements (multiply by boolean result of test).
906  // Thank to Laurent Alonso for this idea.
907  const signed_index_t* T = &(cell_to_v_store_[4 * t]);
908 
909  signed_index_t lv1 =
910  (T[1] == v1) | ((T[2] == v1) * 2) | ((T[3] == v1) * 3);
911 
912  signed_index_t lv2 =
913  (T[1] == v2) | ((T[2] == v2) * 2) | ((T[3] == v2) * 3);
914 
915  geo_debug_assert(lv1 != 0 || T[0] == v1);
916  geo_debug_assert(lv2 != 0 || T[0] == v2);
917  geo_debug_assert(lv1 >= 0);
918  geo_debug_assert(lv2 >= 0);
919  geo_debug_assert(lv1 != lv2);
920 
921  f12 = index_t(halfedge_facet_[lv1][lv2]);
922  f21 = index_t(halfedge_facet_[lv2][lv1]);
923  }

958  bool tet_is_conflict(index_t t, const double* p) const {
959 
960  // Lookup tetrahedron vertices
961  const double* pv[4];
962  for(index_t i=0; i<4; ++i) {
963  signed_index_t v = tet_vertex(t,i);
964  pv[i] = (v == -1) ? nullptr : vertex_ptr(index_t(v));
965  }
966 
967  // Check for virtual tetrahedra (then in_sphere()
968  // is replaced with orient3d())
969  for(index_t lf = 0; lf < 4; ++lf) {
970 
971  if(pv[lf] == nullptr) {
972 
973  // Facet of a virtual tetrahedron opposite to
974  // infinite vertex corresponds to
975  // the triangle on the convex hull of the points.
976  // Orientation is obtained by replacing vertex lf
977  // with p.
978  pv[lf] = p;
979  Sign sign = PCK::orient_3d(pv[0],pv[1],pv[2],pv[3]);
980 
981  if(sign > 0) {
982  return true;
983  }
984 
985  if(sign < 0) {
986  return false;
987  }
988 
989  // If sign is zero, we check the real tetrahedron
990  // adjacent to the facet on the convex hull.
991  geo_debug_assert(tet_adjacent(t, lf) >= 0);
992  index_t t2 = index_t(tet_adjacent(t, lf));
993  geo_debug_assert(!tet_is_virtual(t2));
994 
995  // If t2 is already chained in the conflict list,
996  // then it is conflict
997  if(tet_is_in_list(t2)) {
998  return true;
999  }
1000 
1001  // If t2 is marked, then it is not in conflict.
1002  if(tet_is_marked(t2)) {
1003  return false;
1004  }
1005 
1006  return tet_is_conflict(t2, p);
1007  }
1008  }
1009 
1010  // If the tetrahedron is a finite one, it is in conflict
1011  // if its circumscribed sphere contains the point (this is
1012  // the standard case).
1013 
1014  if(weighted_) {
1015  double h0 = heights_[finite_tet_vertex(t, 0)];
1016  double h1 = heights_[finite_tet_vertex(t, 1)];
1017  double h2 = heights_[finite_tet_vertex(t, 2)];
1018  double h3 = heights_[finite_tet_vertex(t, 3)];
1019  index_t pindex = index_t(
1020  (p - vertex_ptr(0)) / int(vertex_stride_)
1021  );
1022  double h = heights_[pindex];
1023  return (PCK::orient_3dlifted_SOS(
1024  pv[0],pv[1],pv[2],pv[3],p,h0,h1,h2,h3,h
1025  ) > 0) ;
1026  }
1027 
1028  return (PCK::in_sphere_3d_SOS(pv[0], pv[1], pv[2], pv[3], p) > 0);
1029  }

1041  static index_t find_4(const signed_index_t* T, signed_index_t v) {
1042  // The following expression is 10% faster than using
1043  // if() statements. This uses the C++ norm, that
1044  // ensures that the 'true' boolean value converted to
1045  // an int is always 1. With most compilers, this avoids
1046  // generating branching instructions.
1047  // Thank to Laurent Alonso for this idea.
1048  // Note: Laurent also has this version:
1049  // (T[0] != v)+(T[2]==v)+2*(T[3]==v)
1050  // that avoids a *3 multiply, but it is not faster in
1051  // practice.
1052  index_t result = index_t(
1053  (T[1] == v) | ((T[2] == v) * 2) | ((T[3] == v) * 3)
1054  );
1055  // Sanity check, important if it was T[0], not explicitly
1056  // tested (detects input that does not meet the precondition).
1057  geo_debug_assert(T[result] == v);
1058  return result;
1059  }
