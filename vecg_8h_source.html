476  template <index_t DIM, class T>
477  inline vecng<DIM, T> normalize(
478  const vecng<DIM, T>& v
479  ) {
480  T s = length(v);
481  if(s > 1e-30) {
482  s = T(1) / s;
483  }
484  return s * v;
485  }
