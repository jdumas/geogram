198  bool halfedge_is_valid(const Halfedge& H) const {
199  return
200  H.facet != Halfedge::NO_FACET &&

215  bool halfedge_is_border(const Halfedge& H) const {
216  geo_debug_assert(halfedge_is_valid(H));
217  if(facet_region_.is_bound()) {
218  index_t f = H.facet;
219  index_t adj_f =
220  mesh_.facet_corners.adjacent_facet(H.corner);
221  return
222  adj_f == NO_FACET ||
223  facet_region_[f] != facet_region_[adj_f]
224  ;
225  }
226  return mesh_.facet_corners.adjacent_facet(H.corner) == NO_FACET;
227  }
