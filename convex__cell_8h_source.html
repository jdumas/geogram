378  inline double det3x3(
379  double a11, double a12, double a13,
380  double a21, double a22, double a23,
381  double a31, double a32, double a33
382  ) {
383  return
384  a11*det2x2(a22,a23,a32,a33)
385  -a21*det2x2(a12,a13,a32,a33)
386  +a31*det2x2(a12,a13,a22,a23);
387  }
388 
389  inline double det4x4(
390  double a11, double a12, double a13, double a14,
391  double a21, double a22, double a23, double a24,
392  double a31, double a32, double a33, double a34,
393  double a41, double a42, double a43, double a44
394  ) {
395  double m12 = a21*a12 - a11*a22;
396  double m13 = a31*a12 - a11*a32;
397  double m14 = a41*a12 - a11*a42;
398  double m23 = a31*a22 - a21*a32;
399  double m24 = a41*a22 - a21*a42;
400  double m34 = a41*a32 - a31*a42;
401 
402  double m123 = m23*a13 - m13*a23 + m12*a33;
403  double m124 = m24*a13 - m14*a23 + m12*a43;
404  double m134 = m34*a13 - m14*a33 + m13*a43;
405  double m234 = m34*a23 - m24*a33 + m23*a43;
406 
407  return (m234*a14 - m134*a24 + m124*a34 - m123*a44);
408  }
