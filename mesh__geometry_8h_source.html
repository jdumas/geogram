165  inline double mesh_facet_area(const Mesh& M, index_t f, index_t dim=0) {
166  geo_debug_assert(dim <= M.vertices.dimension());
167  if(dim == 0) {
168  dim = M.vertices.dimension();
169  }
170  double result = 0.0;
171  // Check for empty facet, should not happen.
172  if(M.facets.corners_end(f) == M.facets.corners_begin(f)) {
173  return result;
174  }
175  const double* p0 = M.vertices.point_ptr(
176  M.facet_corners.vertex(M.facets.corners_begin(f))
177  );
178  for(
179  index_t i = M.facets.corners_begin(f) + 1;
180  i + 1 < M.facets.corners_end(f); i++
181  ) {
182  result += GEO::Geom::triangle_area(
183  p0,
184  M.vertices.point_ptr(M.facet_corners.vertex(i)),
185  M.vertices.point_ptr(M.facet_corners.vertex(i + 1)),
186  coord_index_t(dim)
187  );
188  }
189  return result;
190  }

208  inline vec3 mesh_facet_center(const Mesh& M, index_t f) {
209  vec3 result(0.0, 0.0, 0.0);
210  double count = 0.0;
211  for(index_t c = M.facets.corners_begin(f);
212  c < M.facets.corners_end(f); ++c) {
213  result += Geom::mesh_corner_vertex(M, c);
214  count += 1.0;
215  }
216  return (1.0 / count) * result;
217  }

226  inline vec3 mesh_tet_center(const Mesh& M, index_t t) {
227  index_t iv1 = M.cells.vertex(t, 0);
228  index_t iv2 = M.cells.vertex(t, 1);
229  index_t iv3 = M.cells.vertex(t, 2);
230  index_t iv4 = M.cells.vertex(t, 3);
231  const vec3& v1 = Geom::mesh_vertex(M, iv1);
232  const vec3& v2 = Geom::mesh_vertex(M, iv2);
233  const vec3& v3 = Geom::mesh_vertex(M, iv3);
234  const vec3& v4 = Geom::mesh_vertex(M, iv4);
235  return 0.25 * (v1 + v2 + v3 + v4);
236  }

247  inline vec3 mesh_corner_vector(const Mesh& M, index_t c1) {
248  geo_debug_assert(M.facets.are_simplices());
249  index_t c2 = M.facets.next_corner_around_facet(c1/3, c1);
250  index_t v1 = M.facet_corners.vertex(c1);
251  index_t v2 = M.facet_corners.vertex(c2);
252  return mesh_vertex(M,v2) - mesh_vertex(M,v1);
253  }
