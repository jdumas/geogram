99  index_t make_periodic_vertex(index_t real, index_t instance) const {
100  geo_debug_assert(real < nb_vertices_non_periodic_);
101  geo_debug_assert(instance < 27);
102  return real + nb_vertices_non_periodic_*instance;

110  static index_t T_to_instance(int Tx, int Ty, int Tz) {
111  geo_debug_assert(Tx >= -1 && Tx <= 1);
112  geo_debug_assert(Ty >= -1 && Ty <= 1);
113  geo_debug_assert(Tz >= -1 && Tz <= 1);
114  int i = (Tz+1) + 3*(Ty+1) + 9*(Tx+1);
115  geo_debug_assert(i >= 0 && i < 27);
116  return index_t(reorder_instances[i]);
117  }

124  void periodic_vertex_get_T(index_t pv, int& Tx, int& Ty, int& Tz) const {
125  geo_debug_assert(pv < nb_vertices_non_periodic_ * 27);
126  index_t instance = periodic_vertex_instance(pv);
127  Tx = translation[instance][0];
128  Ty = translation[instance][1];
129  Tz = translation[instance][2];
130  }
131 
137  void periodic_vertex_set_T(index_t& pv, int Tx, int Ty, int Tz) const {
138  geo_debug_assert(pv < nb_vertices_non_periodic_ * 27);
139  geo_debug_assert(Tx >= -1 && Tx <= 1);
140  geo_debug_assert(Ty >= -1 && Ty <= 1);
141  geo_debug_assert(Tz >= -1 && Tz <= 1);
142  pv = make_periodic_vertex(
143  periodic_vertex_real(pv), T_to_instance(Tx, Ty, Tz)
144  );
145  }

153  std::string binary_to_string(index_t m) const {
154  std::string s(32,' ');
155  for(index_t i=0; i<32; ++i) {
156  s[i] = ((m & (1u << (31u-i))) != 0) ? '1' : '0';
157  }
158  return s;
159  }
