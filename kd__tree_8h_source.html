135  NearestNeighbors(
136  index_t nb_neighbors_in,
137  index_t* user_neighbors_in,
138  double* user_neighbors_sq_dist_in,
139  index_t* work_neighbors_in,

141  ) :
142  nb_neighbors(0),

144  neighbors(work_neighbors_in),

146  user_neighbors(user_neighbors_in),
147  user_neighbors_sq_dist(user_neighbors_sq_dist_in),
148  nb_visited(0)
149  {
150  // Yes, '<=' because we got space for n+1 neigbors
151  // in the work arrays.
152  for(index_t i = 0; i <= nb_neighbors; ++i) {
153  neighbors[i] = index_t(-1);
154  neighbors_sq_dist[i] = Numeric::max_float64();
155  }
156  }

162  double furthest_neighbor_sq_dist() const {
163  return
164  nb_neighbors == nb_neighbors_max ?
165  neighbors_sq_dist[nb_neighbors - 1] :
166  Numeric::max_float64()
167  ;
168  }

179  void insert(
180  index_t neighbor, double sq_dist
181  ) {
182  geo_debug_assert(
183  sq_dist <= furthest_neighbor_sq_dist()
184  );
185 
186  int i;
187  for(i=int(nb_neighbors); i>0; --i) {
188  if(neighbors_sq_dist[i - 1] < sq_dist) {
189  break;
190  }
191  neighbors[i] = neighbors[i - 1];
192  neighbors_sq_dist[i] = neighbors_sq_dist[i - 1];
193  }
194 
195  neighbors[i] = neighbor;
196  neighbors_sq_dist[i] = sq_dist;
197 
198  if(nb_neighbors < nb_neighbors_max) {
199  ++nb_neighbors;
200  }
201  }
202 
210  void copy_from_user() {
211  for(index_t i=0; i<nb_neighbors_max; ++i) {
212  neighbors[i] = user_neighbors[i];
213  neighbors_sq_dist[i] = user_neighbors_sq_dist[i];
214  }
215  neighbors[nb_neighbors_max] = index_t(-1);
216  neighbors_sq_dist[nb_neighbors_max] = Numeric::max_float64();
217  nb_neighbors = nb_neighbors_max;
218  }
219 
226  void copy_to_user() {
227  for(index_t i=0; i<nb_neighbors_max; ++i) {
228  user_neighbors[i] = neighbors[i];
229  user_neighbors_sq_dist[i] = neighbors_sq_dist[i];
230  }
231  }
